// URL da Lista de canais original (Região MENA) const M3U_URL = 'https://iptv-org.github.io/iptv/regions/mena.m3u'; // Servidores proxy alternativos para contornar restrições de CORS na TV Box const PROXIES = [ 'https://api.allorigins.win/raw?url=', 'https://corsproxy.io/?url=', '' // Conexão direta como último recurso ]; let proxyIndex = 0; let allChannels = []; let favorites = JSON.parse(localStorage.getItem('inet_favs_v2')) || []; let currentTab = 'all'; let focusedIndex = 0; let hls = null; let selectedUrl = null; let longPressTimer = null; let lastEnterTime = 0; // Registo temporal para detectar duplo clique const LONG_PRESS_DURATION = 5000; // Tempo de ativação de favoritos: 5 segundos const video = document.getElementById('video-player'); const listContainer = document.getElementById('channel-list'); const searchInput = document.getElementById('search-input'); const statusText = document.getElementById('status-text'); const progressContainer = document.getElementById('fav-progress-container'); const progressBar = document.getElementById('fav-progress-bar'); window.onload = () => { loadChannels(); setupInteractions(); }; // Realiza o carregamento da lista de canais contornando bloqueios CORS async function loadChannels() { statusText.innerText = "A Ligar..."; const proxy = PROXIES[proxyIndex]; const url = proxy ? proxy + encodeURIComponent(M3U_URL) : M3U_URL; try { const res = await fetch(url); if (!res.ok) throw new Error("Erro na resposta de rede"); const text = await res.text(); if (text.includes('#EXTM3U')) { parseM3U(text); } else { throw new Error("Formato de lista M3U inválido"); } } catch (e) { console.warn("Falha ao carregar com proxy " + proxyIndex + ". A testar o próximo..."); if (proxyIndex < PROXIES.length - 1) { proxyIndex++; loadChannels(); } else { statusText.innerText = "Erro ao Carregar Lista"; } } } // Processa o conteúdo do ficheiro M3U e extrai os canais function parseM3U(content) { allChannels = []; const lines = content.split('\n'); let current = null; for (let line of lines) { line = line.trim(); if (line.startsWith('#EXTINF:')) { const name = line.split(',').pop().trim(); const logo = line.match(/tvg-logo="([^"]+)"/)?.[1] || ''; current = { name, logo }; } else if (line.startsWith('http') && current) { current.url = line; allChannels.push(current); current = null; } } if (allChannels.length > 0) { statusText.innerText = allChannels.length + " Canais Carregados"; renderChannels(); focusedIndex = 2; // Inicia o foco no primeiro card de canal disponível updateFocus(); } else { statusText.innerText = "Nenhum Canal Encontrado"; } } // Renderiza a lista de canais de acordo com a aba e pesquisa atual function renderChannels() { const query = searchInput.value.toLowerCase(); let base = currentTab === 'all' ? allChannels : allChannels.filter(c => favorites.includes(c.url)); const filtered = base.filter(c => c.name.toLowerCase().includes(query)); if (filtered.length === 0) { listContainer.innerHTML = `

Nenhum canal localizado.

`; return; } listContainer.innerHTML = filtered.map((c, idx) => { const isFav = favorites.includes(c.url); return `

${c.name}

`}).join(''); } // Configuração dos eventos de teclado (comando físico da box) e interações function setupInteractions() { document.addEventListener('keydown', (e) => { const focusables = document.querySelectorAll('.focusable'); const isCinema = document.body.classList.contains('cinema-mode'); if ([37, 38, 39, 40].includes(e.keyCode)) e.preventDefault(); switch(e.keyCode) { case 38: // Seta Para Cima if (!isCinema && focusedIndex > 0) { focusedIndex--; updateFocus(); } break; case 40: // Seta Para Baixo if (!isCinema && focusedIndex < focusables.length - 1) { focusedIndex++; updateFocus(); } break; case 37: // Seta Para a Esquerda (Muda para aba Todos) if (!isCinema) { document.getElementById('btn-all').click(); } break; case 39: // Seta Para a Direita (Muda para aba Favoritos) if (!isCinema) { document.getElementById('btn-favs').click(); } break; case 13: // Tecla OK ou Enter do teclado if (isCinema) { toggleCinema(false); return; } // DETECÇÃO DE DUPLO CLIQUE (Dbl Enter) const now = Date.now(); if (now - lastEnterTime < 450) { const item = focusables[focusedIndex]; if (item && item.classList.contains('channel-card')) { toggleCinema(true); if (longPressTimer) { clearInterval(longPressTimer); longPressTimer = null; progressContainer.style.display = 'none'; } lastEnterTime = now; return; } } lastEnterTime = now; if (!longPressTimer) startLongPress(); break; case 8: case 27: // Tecla de Retroceder (Backspace) ou Esc if (isCinema) { toggleCinema(false); e.preventDefault(); } break; } }); document.addEventListener('keyup', (e) => { if (e.keyCode === 13) { endLongPress(); } }); // Seleção de Abas document.getElementById('btn-all').onclick = () => { currentTab = 'all'; document.getElementById('btn-all').classList.add('text-blue-400'); document.getElementById('btn-favs').classList.remove('text-blue-400'); renderChannels(); focusedIndex = 2; updateFocus(); }; document.getElementById('btn-favs').onclick = () => { currentTab = 'favs'; document.getElementById('btn-favs').classList.add('text-blue-400'); document.getElementById('btn-all').classList.remove('text-blue-400'); renderChannels(); focusedIndex = 2; updateFocus(); }; searchInput.addEventListener('input', () => { renderChannels(); }); listContainer.addEventListener('click', (e) => { const card = e.target.closest('.channel-card'); if (card) { const url = card.dataset.url; handleSelect(url); } }); listContainer.addEventListener('dblclick', (e) => { const card = e.target.closest('.channel-card'); if (card) { toggleCinema(true); } }); document.getElementById('player-area').addEventListener('click', () => { const isCinema = document.body.classList.contains('cinema-mode'); toggleCinema(!isCinema); }); } // --- SISTEMA PREMUTA DE FAVORITOS (OK PRESSIONADO POR 5 SEGUNDOS) --- function startLongPress() { progressContainer.style.display = 'block'; progressBar.style.width = '0%'; let start = Date.now(); longPressTimer = setInterval(() => { let elapsed = Date.now() - start; let pct = (elapsed / LONG_PRESS_DURATION) * 100; progressBar.style.width = Math.min(pct, 100) + '%'; if (elapsed >= LONG_PRESS_DURATION) { clearInterval(longPressTimer); longPressTimer = "DONE"; toggleFavoriteCurrent(); setTimeout(() => { progressContainer.style.display = 'none'; }, 500); } }, 50); } function endLongPress() { if (longPressTimer && longPressTimer !== "DONE") { clearInterval(longPressTimer); handleNormalOK(); } if (longPressTimer !== "DONE") { progressContainer.style.display = 'none'; } longPressTimer = null; } function handleNormalOK() { const focusables = document.querySelectorAll('.focusable'); const element = focusables[focusedIndex]; if (!element) return; if (element.classList.contains('channel-card')) { handleSelect(element.dataset.url); } else { element.click(); if (element === searchInput) searchInput.focus(); } } function toggleFavoriteCurrent() { const focusables = document.querySelectorAll('.focusable'); const element = focusables[focusedIndex]; const urlToFav = element?.dataset?.url || selectedUrl; if (urlToFav) { if (favorites.includes(urlToFav)) { favorites = favorites.filter(f => f !== urlToFav); } else { favorites.push(urlToFav); } localStorage.setItem('inet_favs_v2', JSON.stringify(favorites)); renderChannels(); showInfoHUD(urlToFav); } } function updateFocus() { const items = document.querySelectorAll('.focusable'); items.forEach(c => c.classList.remove('focused')); if (items[focusedIndex]) { items[focusedIndex].classList.add('focused'); items[focusedIndex].scrollIntoView({ block: 'center', behavior: 'smooth' }); } } function handleSelect(url) { if (selectedUrl === url) { const isCinema = document.body.classList.contains('cinema-mode'); toggleCinema(!isCinema); return; } selectedUrl = url; renderChannels(); playStream(url); showInfoHUD(url); } function playStream(url) { if (hls) hls.destroy(); video.pause(); showError(false); if (Hls.isSupported()) { hls = new Hls({ manifestLoadingMaxRetry: 5 }); hls.loadSource(url); hls.attachMedia(video); hls.on(Hls.Events.MANIFEST_PARSED, () => video.play().catch(()=>{})); hls.on(Hls.Events.ERROR, (e, d) => { if (d.fatal) showError(true); }); } else { video.src = url; video.play().catch(() => showError(true)); } } function toggleCinema(active) { if (active) { document.body.classList.add('cinema-mode'); } else { document.body.classList.remove('cinema-mode'); } setTimeout(() => { window.dispatchEvent(new Event('resize')); }, 100); } function showInfoHUD(url) { const channel = allChannels.find(c => c.url === url); if (!channel) return; const overlay = document.getElementById('info-overlay'); const star = document.getElementById('info-star'); document.getElementById('info-name').innerText = channel.name; star.className = `fa fa-star text-2xl ${favorites.includes(url) ? 'star-fav' : 'star-empty'}`; overlay.style.opacity = "1"; setTimeout(() => { overlay.style.opacity = "0"; }, 4000); } function showError(show) { const screen = document.getElementById('error-screen'); if (show) screen.classList.remove('hidden'); else screen.classList.add('hidden'); }
